scModel
service in module smartCourtLib

Description

因为angular.controller是无状态的，在页面每次切换时$scope的属性都会重新初始化。 需要保持的状态要单独保存在service中，scModel就是这样一个Service 使用是需要扩展此服务 

//model.js
app.factory('listModel', ['scModel', function (scModel) {
  var cityModel =  angular.extend({}, scModel, {
    //model,可选，默认结构
    model:{

    }
  }

  return cityModel;
});

//controller.js
module.controller(['cityModel','$http',function(cityModel,$http){
  $scope.model = cityModel.model;

  var promise = $http.get('/testapi');

  promise.then(function(data){
    cityModel.save($scope,'message',data);
  });
}])
