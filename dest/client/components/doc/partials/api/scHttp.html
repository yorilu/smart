scHttp

Description

封装$http,用以访问restful接口的基类，退服务端返回的错误报文也做了处理

用法

var city =  angular.extend({}, scHttp, {
 url: 'getCityList',   //必选，api 接口
 param: '',            //可选，查询参数
 cache: true           //可选，是否使用缓存，默认为false
});


//get方法
var promise = city.get({cityId:1});
promise.then(function(data){
  //成功处理,此时数据只返回数据体
},function(data){
  //失败处理,数据包含服务端错误码
})
